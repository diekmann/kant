    ‚áí ('person ‚áí ('person, 'etwas) abmachung list)"

  apply(simp add: konsensswap_def)

  by(simp add: fun_eq_iff konsensswap_def swap_symmetric)

  "konsensswap p1 p2 kons p =  map (swap p1 p2) (swap p1 p2 kons p)"
