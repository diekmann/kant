definition bevoelkerung :: "'person set" where "bevoelkerung ≡ UNIV"
definition wenn_jeder_so_handelt
    :: "'world ⇒ ('person, 'world) handlungF ⇒ ('world handlung) set"
  where
    "wenn_jeder_so_handelt welt handlung ≡
      (λhandelnde_person. handeln handelnde_person welt handlung) ` bevoelkerung"
fun was_wenn_jeder_so_handelt_aus_sicht_von
    :: "'world ⇒ ('person, 'world) handlungF ⇒ ('person, 'world) maxime ⇒ 'person ⇒ bool"
  where
    "was_wenn_jeder_so_handelt_aus_sicht_von welt handlung (Maxime m) betroffene_person =
        (∀ h ∈ wenn_jeder_so_handelt welt handlung. m betroffene_person h)"

(*Welt in ihrem aktuellen Zustand. TODO: eigentlich sollten wir für jede mögliche Welt testen!*)

definition teste_maxime ::
  "'world ⇒ ('person, 'world) handlungF ⇒ ('person, 'world) maxime ⇒ bool" where
"teste_maxime welt handlung maxime ≡
  ∀p ∈ bevoelkerung. was_wenn_jeder_so_handelt_aus_sicht_von welt handlung maxime p"

lemma teste_maxime_unfold:
  "teste_maxime welt handlung (Maxime m) =
        (∀p∈bevoelkerung. ∀x∈bevoelkerung. m p (handeln x welt handlung))"
  by(simp add: teste_maxime_def wenn_jeder_so_handelt_def)
lemma "teste_maxime welt handlung (Maxime m) =
        (∀(p,x)∈bevoelkerung×bevoelkerung. m p (handeln x welt handlung))"
  unfolding teste_maxime_unfold by simp

text‹Versuch eine executable version zu bauen.
Wir müssen die Bevölkerung enumerieren.›
definition teste_maxime_exhaust where
  "teste_maxime_exhaust bevoelk welt handlung maxime ≡
    (case maxime of (Maxime m) ⇒ 
      list_all (λ(p,x). m p (handeln x welt handlung)) (List.product bevoelk bevoelk))"

lemma teste_maxime_exhaust_univ: "set b = (UNIV::'person set) ⟹
        teste_maxime welt handlung maxime = teste_maxime_exhaust b welt handlung maxime"
  apply(case_tac maxime, rename_tac m, simp)
  unfolding teste_maxime_unfold teste_maxime_exhaust_def bevoelkerung_def
  apply(simp)
  by(simp add: list_all_iff)
