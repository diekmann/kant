section‹Beispiel: Steuern›

text‹Wenn die Welt sich durch eine Zahl darstellen lässt, ...›
datatype steuerwelt = Steuerwelt
        (get_einkommen: "person ⇒ int") ―‹einkommen: einkommen jeder Person (im Zweifel 0).›

fun steuerlast :: "person ⇒ steuerwelt handlung ⇒ int" where
  "steuerlast p (Handlung vor nach) = ((get_einkommen vor) p) - ((get_einkommen nach) p)"

text‹Default: kein Einkommen. Um Beispiele einfacher zu schreiben.›
definition KE :: "person ⇒ int" where
  "KE ≡ λp. 0"

lemma ‹steuerlast Alice (Handlung (Steuerwelt (KE(Alice:=8))) (Steuerwelt (KE(Alice:=5)))) = 3›
  by eval
lemma ‹steuerlast Alice (Handlung (Steuerwelt (KE(Alice:=8))) (Steuerwelt (KE(Alice:=0)))) = 8›
  by eval
lemma ‹steuerlast Bob   (Handlung (Steuerwelt (KE(Alice:=8))) (Steuerwelt (KE(Alice:=5)))) = 0›
  by eval
lemma ‹steuerlast Alice (Handlung (Steuerwelt (KE(Alice:=-3))) (Steuerwelt (KE(Alice:=-4)))) = 1›
  by eval
lemma ‹steuerlast Alice (Handlung (Steuerwelt (KE(Alice:=1))) (Steuerwelt (KE(Alice:=-1)))) = 2›
  by eval

fun mehrverdiener :: "person ⇒ steuerwelt handlung ⇒ person set" where
  "mehrverdiener ich (Handlung vor nach) = {p. (get_einkommen vor) p ≥ (get_einkommen vor) ich}"

lemma ‹mehrverdiener Alice
        (Handlung (Steuerwelt (KE(Alice:=8, Bob:=12, Eve:=7))) (Steuerwelt (KE(Alice:=5))))
       = {Alice, Bob}› by eval

(*TODO: maxime sollte sein, dass ich mehr steuern zu zahlen hab als geringerverdiener.*)
definition maxime_steuern :: "(person, steuerwelt) maxime" where
  "maxime_steuern ≡ Maxime 
      (λich handlung.
           ∀p∈mehrverdiener ich handlung. steuerlast ich handlung ≤ steuerlast p handlung)"



fun delta_steuerwelt :: "(steuerwelt, person, int) delta" where
  "delta_steuerwelt vor nach = Aenderung.delta_num_fun (get_einkommen vor) (get_einkommen nach)"

definition "sc ≡ SimConsts
    Alice
    maxime_steuern
    (printable_case_law_ableiten_absolut (λw. show_fun (get_einkommen w)))"
definition "sc' ≡ SimConsts
    Alice
    maxime_steuern
    (case_law_ableiten_relativ delta_steuerwelt)"

definition "initialwelt ≡ Steuerwelt (KE(Alice:=8, Bob:=3, Eve:= 5))"

definition "beispiel_case_law h ≡ simulateOne sc 3 h initialwelt (Gesetz {})"
definition "beispiel_case_law' h ≡ simulateOne sc' 20 h initialwelt (Gesetz {})"
