fun stehlen4_partial :: ‹int ⇒ int ⇒ person ⇒ zahlenwelt ⇒ zahlenwelt option› where
    ‹stehlen4_partial beute opfer_nach_besitz dieb (Zahlenwelt besitz) =
      (case opfer_eindeutig_nach_besitz_auswaehlen opfer_nach_besitz besitz Enum.enum
         of None ⇒ None
          | Some opfer ⇒ if opfer = dieb
                          then None
                          else Some (Zahlenwelt ⟦⟦besitz(opfer -= beute)⟧(dieb += beute)⟧)
      )›
